choice=int(input("\n 1.Add \n 2.Substract \n 3.Divide \n 4.Multiply\n"))
a=int(input("Enter a number Value : "))
b=int(input("Enter a number Value : "))

if choice==1:
    print(a+b)
elif choice==2:
    print(a-b)
elif choice==3:
    print(a/b)
elif choice==4:
    print(a*b)
else :
    print("You Entered Wrong Input")
